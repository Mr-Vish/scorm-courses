Automated Document Remediation Pipeline

import fitz  # PyMuPDF
from openai import OpenAI

client = OpenAI()

def remediate_pdf(pdf_path: str) -> dict:
    '''Analyze a PDF and generate accessibility remediation.'''
    doc = fitz.open(pdf_path)
    remediation = {"pages": []}

    for page_num, page in enumerate(doc):
        # Extract text and images
        text = page.get_text()
        images = page.get_images()

        # Generate heading structure
        heading_response = client.chat.completions.create(
            model="gpt-4o-mini",
            response_format={"type": "json_object"},
            messages=[{
                "role": "user",
                "content": (
                    "Analyze this page text and identify the heading hierarchy. "
                    "Return JSON: {"headings": [{"level": int, "text": str, "line": int}]}

"
                    f"Page text:
{text}"
                ),
            }],
        )

        page_info = {
            "page": page_num + 1,
            "headings": json.loads(heading_response.choices[0].message.content),
            "images_needing_alt": len(images),
            "has_tables": "table" in text.lower() or "|" in text,
        }
        remediation["pages"].append(page_info)

    return remediation

Simplification for Cognitive Accessibility

def simplify_text(text: str, reading_level: str = "8th grade") -> str:
    '''Simplify text for cognitive accessibility.'''
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "system", "content": (
                f"Rewrite the following text at a {reading_level} reading level. "
                "Use short sentences. Use common words. "
                "Break complex ideas into simple steps. "
                "Keep the same meaning and key information."
            )},
            {"role": "user", "content": text},
        ],
    )
    return response.choices[0].message.content

Best Practices

  • AI as assistant, not replacement: AI-generated alt text should be reviewed by humans, especially for critical content
  • Test with real users: Involve people with disabilities in testing AI-generated accessibility features
  • Progressive enhancement: Start with the highest-impact issues (images, headings) before addressing everything
  • Maintain context: Provide page context to the AI so alt text and descriptions are relevant to the surrounding content
  • Monitor quality: Regularly audit AI-generated accessibility content for accuracy and helpfulness
